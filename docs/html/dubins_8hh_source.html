158  Point2<T> pi,
159  Point2<T> pf,
160  Point2<T> p

162 {
163  //Compute circonference with Cramer
164  double d0=-((pow2(p0.x())+pow2(p0.y())));
165  double di=-((pow2(pi.x())+pow2(pi.y())));
166  double df=-((pow2(pf.x())+pow2(pf.y())));
167  //Compute determinants
168  double D = (p0.x()*pi.y())+(p0.y()*pf.x())+(pi.x()*pf.y()) - ( p0.x()*pf.y()+p0.y()*pi.x()+(pi.y()*pf.x()) );
169  double Da= (p0.x()*d0)+(p0.y()*df)+(pf.y()*di) - ( d0*pf.y()+di*p0.y()+df*pi.y() );
170  double Db= (p0.x()*di)+(d0*pf.x())+(pi.x()*df) - ( df*p0.x()+ d0*pi.x()+di*pf.x() );
171  double Dc= (p0.x()*pi.y()*df)+(p0.y()*di*pf.x())+(d0*pi.x()*pf.y()) - ( p0.x()*di*pf.y()+p0.y()*pi.x()*df+d0*pi.y()*pf.x() );
172  //Compute circle's parameters
173  double a=Da/D;
174  double b=Db/D;
175  double c=Dc/D;
176 
177  bool ok=true;
178 
179  if ( equal((pow2(p.x())+pow2(p.y())+a*p.x()+b*p.y()+c), 0.0) ){ //Check if point is on circonference
180  Point2<double> center (-a/2.0, -b/2.0); //Compute center
181 
182  //Compute angles of extremities and point of intereset p. Mind that center.th(p0) is 0 only if p0 is on the right.
183  Angle th0=center.th(p0);
184  Angle thf=center.th(pf);
185  Angle th=center.th(p);
186 
187  if (th>=th0 && th<=thf){ //Point is not in the arc.
188  ok=false;
189  }
190  }
191  else {
192  ok=false;
193  }
